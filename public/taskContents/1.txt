let result={}

    data.replace(/([~!@#$%^&*()_+=`{}\[\]\|\\:;'<>\s",.\/? ])+/g, '-').replace(/^(-)+|(-)+$/g,'').split('-').forEach( word => {
        word = word.replace(/\s/g, "");
        if(word !== " " && word !== ""){
            if(result[word]){
                result[word]=result[word]+1
            }else{
                result[word]=1;
            }
        }
    });
return result;
