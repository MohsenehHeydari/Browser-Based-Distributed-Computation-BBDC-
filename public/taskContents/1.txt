let result={}

    data.replace(/([~!@#$%^&*()_+=`{}\[\]\|\\:;'<>",.\/? ])+/g, '-').replace(/^(-)+|(-)+$/g,'').split('-').forEach( word => {
        if(word !== " " && word !== ""){
            if(result[word]){
                result[word]=result[word]+1
            }else{
                result[word]=1;
            }
        }
    });
return result;
